\section{Homology Theory}
Consider the functor $h_*: \{\text{Pairs of top space} (X, A)\} \to Ab$ which maps morphisms
\begin{equation*}
	h_*(f: (X,A) \to (Y,B)) = h_*(X,A) \to h_*(Y,B).
\end{equation*}

\textbf{Axioms of Homology}
\begin{itemize}
	\item Dimension; $X = \{pt\}$, $h_n(X) = 0, \forall n$.
	\item Exactness; $A \to i$, $(X, \empyset) \to (X, A)$; $\cdots \to h_n(A) \to i_* \cdots.$
	\item Excision;  $X = A \cup B$ with $(A,B, X top)$ then $A, A \cap B) \to (X, B)$ induces an isomorphism $h_*(A, A \cap B) \to h_*(X, B).$ 
\end{itemize}

\textbf{Fact 1.} Singular homology satisfies ES Axioms\\
\textbf{Fact 2.} Mayer-Viatoris Sequence \\

\section{Applications of Axioms}
\textbf{Fact.} $(X,A) \simeq (Y,B)$ implies $H_*(X, A) \simeq H_*(Y,B).$ 
\textbf{Reduced Homology.} $X = \emptyset$, $\epsilon: X \to \{pt\}.$ $\epsilon_*: H_*(X) \to H_*(\{pt\}).$
